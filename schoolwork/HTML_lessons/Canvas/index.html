Canvas lesson

Description
How to draw your first Canvas? Make it look like a foggy day.
Solution
Code Example 
      var cc = document.getElementById('myCanvas2');
      var ctx = cc.getContext('2d');

      ctx.shadowBlur = 20;
      ctx.shadowColor = "black";

      var grd = ctx.createLinearGradient(0, 250, 0, 0);
      grd.addColorStop(0, "rgb(156, 170, 180)");
      grd.addColorStop(1, "lightblue");
      ctx.fillStyle = grd;
      ctx.fillRect(0, 0, 400, 150);
